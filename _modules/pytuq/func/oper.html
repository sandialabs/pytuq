    def grad(self, x):

        grad = np.empty((x.shape[0], self.outdim, self.dim))
        for i in range(x.shape[0]):
            grad[i, :, :] = np.dot(self.fcn2.grad(self.fcn1(x))[i], self.fcn1.grad(x)[i])
        return grad

        """Function call.
