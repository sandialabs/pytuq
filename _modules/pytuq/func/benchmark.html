        self.a = np.array([(i-2.)/2. for i in range(1,self.dim+1)])




    def __call__(self, x):
        sam = x.shape[0]
        self.checkDim(x)

        ydata=np.empty((sam, self.outdim))
        for j in range(sam):
            val=1.
            for k in range(self.dim):
                val *= ( (abs(4.*x[j,k]-2.)+self.a[k])/(1.+self.a[k]) )
            ydata[j,0]=val

        return ydata

        -------
        numpy array, 1dim
